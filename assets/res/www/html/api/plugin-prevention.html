var result = M.plugin('prevention').rooting();

if ( result.status === 'SUCCESS' ) {
    if ( result.rooting === true ) {
    console.log( '루팅폰 입니다.' )
    }
    else {
    console.log( '루팅폰이 아닙니다' );
    }
}
else {
    if ( result.error ) {
    console.log( result.error );
    }
    else {
    console.log( JSON.stringify( result ) );
    }
}

M.plugin('prevention').fraud(
{
    callback : function(event) {
        console.log("event::"+event);
        if (event.status === 'SUCCESS') {
            if (event.rooting === true) {
                console.log('루팅폰 입니다.')
            } else {
                if (event.valid === false) {
                    alert('앱이 유효하지 않습니다\n 계속 실행할 수 없어 종료합니다.');
                    //M.sys.exit();
                } else {
                    console.log('유효한 앱입니다.');
                    M.page.html("main.html", {
                        delay : 500,
                        action : "NEW_SCR",
                        orient : "PORT",
                        animation : "DEFAULT"
                    });
                }
            }
        } else {
            if (event.error) {
                console.log("error::"+event.error);
                alert('앱이 유효하지 않습니다\n 계속 실행할 수 없어 종료합니다.');
                //M.sys.exit();
            } else {
                alert("alert::"+JSON.stringify(event));
            }
        }
    }
});
